<script>
pdfjsLib.GlobalWorkerOptions.workerSrc='https://cdnjs.cloudflare.com/ajax/libs/pdf.js/3.11.174/pdf.worker.min.js';

// ===== 楽器 =====
const MAIN=[
{l:"Pic",c:"#e35"},{l:"Fl",c:"#e35"},{l:"Ob",c:"#c44"},
{l:"B♭Cl",c:"#36a"},{l:"B.Cl",c:"#36a"},
{l:"A.Sax",c:"#1a8"},{l:"T.Sax",c:"#1a8"},{l:"B.Sax",c:"#1a8"},
{l:"Tp",c:"#c70"},{l:"Tb",c:"#a84"},
{l:"Hr",c:"#84a"},{l:"B.Tb",c:"#a84"},
{l:"Tu",c:"#08a"},{l:"Eu",c:"#08a"},{l:"Stb",c:"#555"}
];

const OTHER=[
{l:"Cor Angl.",c:"#c44"},
{l:"E♭Cl",c:"#36a"},
{l:"A.Cl",c:"#36a"},
{l:"CA.Cl",c:"#36a"},
{l:"CB.Cl",c:"#36a"},
{l:"S.Sax",c:"#1a8"},
{l:"PicTp",c:"#c70"},
{l:"CorTp",c:"#c70"}
];

let currentInst=MAIN[0];

function build(list,id){
 let el=document.getElementById(id);
 list.forEach(i=>{
  let d=document.createElement("div");
  d.className="inst";
  d.textContent=i.l;
  d.style.color=i.c;
  d.onclick=()=>{
    currentInst=i;
    document.querySelectorAll(".inst").forEach(x=>x.classList.remove("active"));
    d.classList.add("active");
  };
  el.appendChild(d);
 });
}

build(MAIN,"mainList");
build(OTHER,"otherList");
document.querySelector(".inst").classList.add("active");

function switchTab(n,el){
 document.querySelectorAll(".tab").forEach(t=>t.classList.remove("active"));
 el.classList.add("active");
 document.getElementById("mainList").style.display=n==="main"?"":"none";
 document.getElementById("otherList").style.display=n==="other"?"":"none";
}

// ===== core =====
const GRID=30;
let elements=[];
let tool="select";
let id=0;
let arcPts=[];

function snap(v){return Math.round(v/GRID)*GRID;}

function addChair(x,y){
 elements.push({x,y,label:currentInst.l,color:currentInst.c});
}

// ===== render =====
function render(){
 stage.querySelectorAll(".elem").forEach(e=>e.remove());
 elements.forEach(el=>{
  let d=document.createElement("div");
  d.className="elem";
  d.style.left=el.x+"px";
  d.style.top=el.y+"px";

  let c=document.createElement("div");
  c.className="chair";
  c.textContent=el.label;
  c.style.borderColor=el.color;
  c.style.color=el.color;

  d.appendChild(c);
  stage.appendChild(d);
 });
}

// ===== 弧 =====
function bez(p0,p1,p2,t){
 return {
  x:(1-t)*(1-t)*p0.x+2*(1-t)*t*p1.x+t*t*p2.x,
  y:(1-t)*(1-t)*p0.y+2*(1-t)*t*p1.y+t*t*p2.y
 };
}

function placeArc(){
 let n=parseInt(document.getElementById("arcN").value)||8;
 for(let i=0;i<n;i++){
  let t=i/(n-1);
  let p=bez(arcPts[0],arcPts[1],arcPts[2],t);
  addChair(p.x-13,p.y-13);
 }
 arcPts=[];
 render();
}

// ===== stage =====
stage.onclick=e=>{
 let r=stage.getBoundingClientRect();
 let x=snap(e.clientX-r.left);
 let y=snap(e.clientY-r.top);

 if(tool==="chair"){addChair(x,y);render();}

 if(tool==="arc"){
  arcPts.push({x,y});
  if(arcPts.length===3) placeArc();
 }
};

// ===== 読み込み =====
function loadPDF(){
 let i=document.createElement("input");i.type="file";i.accept=".pdf";
 i.onchange=e=>{
  let f=e.target.files[0];
  let r=new FileReader();
  r.onload=ev=>{
   pdfjsLib.getDocument({data:new Uint8Array(ev.target.result)}).promise.then(pdf=>{
    pdf.getPage(1).then(p=>{
     let v=p.getViewport({scale:1});
     let c=document.getElementById("bg");
     let ctx=c.getContext("2d");
     c.width=v.width;c.height=v.height;
     p.render({canvasContext:ctx,viewport:v});
    });
   });
  };
  r.readAsArrayBuffer(f);
 };
 i.click();
}

function loadImg(){
 let i=document.createElement("input");i.type="file";
 i.onchange=e=>{
  let img=new Image();
  img.onload=()=>{
   let c=document.getElementById("bg");
   let ctx=c.getContext("2d");
   c.width=img.width;c.height=img.height;
   ctx.drawImage(img,0,0);
  };
  img.src=URL.createObjectURL(e.target.files[0]);
 };
 i.click();
}

// ===== export =====
function exportPNG(){
 html2canvas(stage).then(c=>{
  let a=document.createElement("a");
  a.href=c.toDataURL();
  a.download="stage.png";
  a.click();
 });
}

function setTool(t){
 tool=t;
 document.getElementById("arcUI").style.display=(t==="arc")?"block":"none";
}

render();
